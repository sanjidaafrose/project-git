package avg;

 *
 * @author SEU
 */
public class Avg {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        // TODO code application logic here
        int[] a = new int[50];
        double avg = 0;
        for (int i = 0; i < 50; i++) {
            a[i] = i + 1;
        }
        for (int i = 0; i < 50; i++) {
            avg += a[i];
        }
        System.out.println( "Avg value is: " + (avg / 50));
    }

}
